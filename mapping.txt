# ==================== DCS 戰術詞彙外掛對照網 ====================

# 1. 徹底剿滅 take-off 和 start-up 的連字號與空格
take-off = takeoff
take off = takeoff
# 強制剿滅耳機悶音導致的同音聽錯（Acoustic Correction）
tick off = takeoff
tick-off = takeoff

start-up = startup
start up = startup

# 2. 徹底剿滅 requests 帶尾音 s 的變體

requests landing = request landing
requests taxi = request taxi
